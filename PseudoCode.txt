FUNCTION capital(input: STRING): STRING
    IF input IS NULL OR input IS EMPTY THEN
        RETURN input
    END IF
    RETURN FIRST_CHARACTER(input) IN UPPERCASE + REMAINING_CHARACTERS(input) IN LOWERCASE
END FUNCTION

MAIN
    INITIALIZE scanner TO READ INPUT
    INITIALIZE incidents AS EMPTY LIST OF STRING ARRAYS

// Input data
    WHILE scanner HAS NEXT LINE
        READ line FROM scanner AND TRIM WHITESPACE
        IF line IS EMPTY THEN
            BREAK
        END IF
        SPLIT line INTO names ARRAY BY SPACE
        IF names LENGTH IS EQUAL TO 2 THEN
            names[0] = capital(names[0])
            names[1] = capital(names[1])
            ADD names TO incidents
        END IF
    END WHILE
    CLOSE scanner

    INITIALIZE killerCount AS EMPTY MAP OF STRING TO INTEGER
    INITIALIZE victims AS EMPTY SET OF STRING

// Memproses setiap insiden
    FOR EACH incident IN incidents
        SET killer TO incident[0]
        SET victim TO incident[1]
        
        IF killer EXISTS IN killerCount THEN
            INCREMENT killerCount[killer] BY 1
        ELSE
            SET killerCount[killer] TO 1
        END IF
        
        ADD victim TO victims
    END FOR

// menghilangkan pembunuh yang juga menjadi korban 
    FOR EACH killer IN killerCount KEYS
        IF killer IS IN victims THEN
            REMOVE killer FROM killerCount
        END IF
    END FOR
  
// mengurutkan pembunuh beradasarkan abjad
    INITIALIZE killers AS LIST OF killerCount KEYS
    SORT killers


// Output
    PRINT "DAFTAR PENCARIAN ORANG KASUS PEMBUNUHAN"
    FOR EACH killer IN killers
        PRINT killer + " " + killerCount[killer]
    END FOR
END MAIN
